Control Flows
Since Angular 17 uses the {{ "@" }} annotation for control flows, if we want to render an email string in the html template we must use {{ htmlAt }} rather than {{ "@" }}

Example if blocks

@if {{ "{}" }} blocks
@if (a > b) { {{ a }} is greater than {{ b }} } @else if (b > a) { {{ a }} is less than {{ b }} } @else { {{ a }} is equal to {{ b }} }
@for {{ "{}" }} blocks
@for (a of alphas; track $index) {
{{ a }} Index: {{ $index }} Count:{{ $count }} Last: {{ $last }} First {{ $first }} Odd: {{ $odd }} Even: {{ $even }}
} @empty { No items available }
@for (o of objs; track o.id) { {{ o.name }}   } @empty { No items available }
@switch {{ "{}" }} blocks
@switch (myColour) { @case ("black") { Black } @case ("green") { Green } @case ("red") { Red } @default { Default case. } }
